184 template <class R, class... Args, size_t sz, size_t algn>
185 struct small_task<R(Args...), sz, algn> {
186  struct vtable_t {
187  void (*mover)(void *src, void *dest);
188  void (*destroyer)(void *);
189  R (*invoke)(void const *t, Args &&... args);
190  template <class T> static vtable_t const *get() {
191  static const vtable_t table = {
192  [](void *src, void *dest) {
193  new (dest) T(move(*static_cast<T *>(src)));
194  },
195  [](void *t) { static_cast<T *>(t)->~T(); },
196  [](void const *t, Args &&... args) -> R {
197  return (*static_cast<T const *>(t))(forward<Args>(args)...);
198  }};
199  return &table;
200  }
201  };
202  vtable_t const *table = nullptr;
203  aligned_storage_t<sz, algn> data;
204  template <
205  class F, class dF = decay_t<F>,
206  enable_if_t<!is_same<dF, small_task>{}> * = nullptr,
207  enable_if_t<is_convertible<res_of_t<dF &(Args...)>, R>{}> * = nullptr>
208  small_task(F &&f) : table(vtable_t::template get<dF>()) {
209  static_assert(sizeof(dF) <= sz, "object too large");
210  static_assert(alignof(dF) <= algn, "object too aligned");
211  new (&data) dF(forward<F>(f));
212  }
213  ~small_task() {
214  if (table)
215  table->destroyer(&data);
216  }
217  small_task(const small_task &o) : table(o.table) {
218  data = o.data;
219  }
220  small_task(small_task &&o) : table(o.table) {
221  if (table)
222  table->mover(&o.data, &data);
223  }
224  small_task() {
225  }
226  small_task &operator=(const small_task &o) {
227  this->~small_task();
228  new (this) small_task(move(o));
229  return *this;
230  }
231  small_task &operator=(small_task &&o) {
232  this->~small_task();
233  new (this) small_task(move(o));
234  return *this;
235  }
236  explicit operator bool() const {
237  return table;
238  }
239  R operator()(Args... args) const {
240  return table->invoke(&data, forward<Args>(args)...);
241  }
242 };
